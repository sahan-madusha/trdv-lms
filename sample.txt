name: QA Branch Deployement

on:
  push:
    branches:
      - qa

jobs:

  web-deploy:
    name: Build React App
    runs-on: ubuntu-latest
    steps:
      - name: Get lates code
        uses: actions/checkout@v3

      - name: Set up Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'

      - name: Install dependencies
        run: |
          cd client
          npm install

      - name: Build the React app
        run: |
          cd client
          npm run build
